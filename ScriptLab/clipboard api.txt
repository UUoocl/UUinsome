    async function getSlideMetadata(direction) {
      //console.log(getActiveFileView());
      await Office.context.document.getSelectedDataAsync(Office.CoercionType.SlideRange, async function(asyncResult) {
        if (asyncResult.status === Office.AsyncResultStatus.Failed) {
          console.error(asyncResult.error.message);
        } else {
          //console.log(JSON.stringify(asyncResult.value, null, 4));
          let result = asyncResult.value;
          result = result.slides[0].index;
          //console.log(`slide index = ${result}`);
          if (direction === "next") {
            await getNotesForOBS(result - 1, "exit").then((pptData) => {
              saveText(JSON.stringify(pptData));
            }).then(() =>{
              goToNextSlide()
            }).then(() => {
              getNotesForOBS(result, "entrance")
            }).then((pptData) => {
                saveText(JSON.stringify(pptData));
            });
            
          }
          if (direction === "previous") {
            await getNotesForOBS(result - 2, "exit").then((pptData) => {
              saveText(JSON.stringify(pptData));
            }).then(() => {
              goToPreviousSlide()
            }).then(() => {
              getNotesForOBS(result, "entrance")
            }).then((pptData) => {
              saveText(JSON.stringify(pptData));
            });
          }


          if (direction === "current") {
            await getNotesForOBS(result-1,"current").then((obj) => {
              console.log("then");
              stringResult = JSON.stringify(obj);
              console.log(`get notes results: ${stringResult}`);
              saveText(stringResult);
              return;
            });

            //document.getElementById("clipboard").click();
            //return;
          }
        }
      });

    // get the "obs-" shapes' Names and Text from the current slide.


    async function getNotesForOBS(slideIndex, action) {
      var pptData = { actionName: `slide_${action}` },
        found = 0;
      const indexArray = [];
      let result = await PowerPoint.run(async (context) => {
        // Get every shape in the current slide.
        const shapes: PowerPoint.ShapeCollection = context.presentation.slides.getItemAt(slideIndex).shapes;
        //load all the shape properties
        shapes.load("items");

        await context.sync();

        const shapeNames = [];
        // find the "|||" shapes index
        await shapes.items.forEach(async (shape, index) => {
           //console.log(shape.name);
           if (shape.name.startsWith("obs-")) {
             found = 1;
             //console.log(`index: ${index}, ID: ${shape.id}, Name: ${shape.name}`);
             indexArray.push(index);
             shapeNames.push(shape.name);
           }
         });

         //get Shape Text
         const shapeTextRanges = [];
         const shapeFills = [];
         if (found) {
           //load text
           for (let i = 0; i < indexArray.length; i++) {
             const shapeTextRange: PowerPoint.TextRange = shapes.getItemAt(indexArray[i]).textFrame.textRange;
             const shapeFill: PowerPoint.ShapeFill = shapes.getItemAt(indexArray[i]).fill;
             shapeTextRanges.push(shapeTextRange);
             shapeTextRanges[i].load("text");
             shapeFills.push(shapeFill);
             shapeFills[i].load("transparency");
           }
         }
          await context.sync();

          //Action to Transparency value
          let actionCode;
          switch(action){
            case 'entrance' || 'exit': actionCode = 0; break; //Entrance
            case 'current': actionCode = 0.01; break;  //Exit
          }
          //build the message
          for (let i = 0; i < shapeTextRanges.length; i++) {
            //console.log(shapeNames[i])
            if (shapeTextRanges[i].text.includes("|||") && shapeFills[i].transparency == actionCode){
            const keyValue = shapeTextRanges[i].text.split("|||");
            //console.log(`tag: ${shapeTextRanges[i].text} ActionCode: ${actionCode} transparecy ${shapeFills[i].transparency === actionCode}`)
            pptData[keyValue[0]] = keyValue[1];
            }
            if(shapeNames[i] === "obs-Notes"){
              pptData['notes'] = shapeTextRanges[i].text;

            }

          }
          //pptData.slidePosition = slideIndex + 1;
          console.log( pptData)

          return pptData;

    })

    return result

    }
